Sayfalar

Ana Sayfa: 1. satırda ‘Bakiye: ….’ 2. satıda ‘A: Yükle B: Çek C: Sıfırla D: İptal’ yazılır. ekrana sığmayan karakterler soldan sağa animasyonla sonsuz kaydırılır.

Yükleme sayfası: 1. satırda ‘Yeni bakiye: …’ 2. Satırda ‘#: Onayla D: İptal’ yazılır

Çekme sayfası: 1. satırda ‘Çekilecek bakiye: …’ 2. Satırda ‘#: Onayla D: İptal’ yazılır

#include <Wire.h>
#include <LiquidCrystal_I2C.h>
#include <MFRC522.h>
#include <Keypad.h>

// RFID
#define SDA_PIN 10
#define RST_PIN 9
MFRC522 rfid(SDA_PIN, RST_PIN);

// LCD
LiquidCrystal_I2C lcd(0x27, 16, 2);

// Keypad
const byte ROWS = 4; // 4 satır
const byte COLS = 4; // 4 sütun
char keys[ROWS][COLS] = {
  {'1', '2', '3', 'A'},
  {'4', '5', '6', 'B'},
  {'7', '8', '9', 'C'},
  {'*', '0', '#', 'D'}
};
byte rowPins[ROWS] = {2, 3, 4, 5};
byte colPins[COLS] = {A0, A1, A2, A3};
Keypad keypad = Keypad(makeKeymap(keys), rowPins, colPins, ROWS, COLS);

// Değişkenler
float bakiye = 0.0;


void setup() {
  // Seri Bağlantı
  Serial.begin(9600);

  // RFID Başlat
  SPI.begin();
  rfid.PCD_Init();

  // LCD Başlat
  lcd.begin();
  lcd.backlight();

  // Başlangıç ekranı
  lcdGuncelle("Sisteme Hosgeldin", "Kart okutunuz...");
}

void loop() {
  // Kart durumu
  if (kartOkundu()) {
    lcdGuncelle("Bakiye:", String(bakiye, 2) + " TL");
    lcd.setCursor(0, 1);
    lcd.print("A: Yükle B: Çek");
    char key = keypad.getKey();
    if (key) {
      switch (key) {
        case 'A': // Yükleme işlemi
          bakiyeIslemleri(true);
          break;
        case 'B': // Çekme işlemi
          bakiyeIslemleri(false);
          break;
        case 'C': // Sıfırlama
          bakiye = 0;
          lcdGuncelle("Bakiye sifirlandi", "");
          delay(2000);
          break;
        case 'D': // Ana sayfaya dönüş
          lcdGuncelle("Ana sayfaya don", "");
          delay(2000);
          break;
      }
    }
  } else {
    lcdGuncelle("Kart Bulunamadi", "Lutfen kart okutun");
    delay(1000);
  }
}

void bakiyeIslemleri(bool yukle) {
  lcdGuncelle(yukle ? "Yeni bakiye: " : "Cekilecek bakiye:", "D: Iptal # Onay");
  String input = "";
  while (true) {
    char key = keypad.getKey();
    if (key) {
      if (key == 'D') {
        lcdGuncelle("Ana Sayfa", "");
        delay(1000);
        return;
      } else if (key == '#') {
        float miktar = input.toFloat();
        if (yukle) {
          bakiye += miktar;
        } else {
          bakiye -= miktar;
        }
        lcdGuncelle("Islem Tamamlandi", "");
        delay(2000);
        return;
      } else {
        input += key;
        lcd.setCursor(14, 0);
        lcd.print(input);
      }
    }
  }
}

void lcdGuncelle(String satir1, String satir2) {
  lcd.clear();
  lcd.setCursor(0, 0);
  lcd.print(satir1);
  lcd.setCursor(0, 1);
  lcd.print(satir2);
  Serial.println(satir1)
  Serial.println(satir2)
  Serial.println("------------------")
}

bool kartOkundu() {
  if (!rfid.PICC_IsNewCardPresent() || !rfid.PICC_ReadCardSerial()) {
    return false; // Kart yok
  }
  return true; // Kart mevcut
}
